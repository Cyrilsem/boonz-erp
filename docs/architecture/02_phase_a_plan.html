-- A3: write_audit_log + helpers
-- Constitutional articles enforced: 7 (append-only) and 8 (every RPC audits)

BEGIN;

CREATE TABLE IF NOT EXISTS public.write_audit_log (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  at          timestamptz NOT NULL DEFAULT now(),
  caller_uid  uuid,
  caller_role text,
  via_rpc     boolean NOT NULL DEFAULT false,
  rpc_name    text,
  op          text NOT NULL,           -- INSERT | UPDATE | DELETE
  table_name  text NOT NULL,
  pk          text,
  before      jsonb,
  after       jsonb
);

CREATE INDEX IF NOT EXISTS idx_wal_at ON public.write_audit_log (at DESC);
CREATE INDEX IF NOT EXISTS idx_wal_table_pk ON public.write_audit_log (table_name, pk);
CREATE INDEX IF NOT EXISTS idx_wal_via_rpc_false ON public.write_audit_log (at DESC) WHERE via_rpc = false;

ALTER TABLE public.write_audit_log ENABLE ROW LEVEL SECURITY;

-- read: only superadmin / manager / finance (role lives in public.user_profiles)
DROP POLICY IF EXISTS wal_select ON public.write_audit_log;
CREATE POLICY wal_select ON public.write_audit_log
  FOR SELECT TO authenticated
  USING (EXISTS (
    SELECT 1 FROM public.user_profiles
     WHERE id = (SELECT auth.uid())
       AND role = ANY (ARRAY['superadmin', 'manager', 'finance'])
  ));

-- insert: any authenticated (we want all writes recorded)
DROP POLICY IF EXISTS wal_insert ON public.write_audit_log;
CREATE POLICY wal_insert ON public.write_audit_log
  FOR INSERT TO authenticated WITH CHECK (true);

-- append-only: no update / delete for anyone
DROP POLICY IF EXISTS wal_no_update ON public.write_audit_log;
CREATE POLICY wal_no_update ON public.write_audit_log FOR UPDATE USING (false);
DROP POLICY IF EXISTS wal_no_delete ON public.write_audit_log;
CREATE POLICY wal_no_delete ON public.write_audit_log FOR DELETE USING (false);

-- helper called from canonical RPCs
CREATE OR REPLACE FUNCTION public.audit_log_write(
  p_op text, p_table text, p_pk text, p_before jsonb, p_after jsonb
) RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
BEGIN
  INSERT INTO public.write_audit_log (caller_uid, caller_role, via_rpc, rpc_name, op, table_name, pk, before, after)
  VALUES (
    auth.uid(),
    (SELECT role FROM public.user_profiles WHERE id = auth.uid()),
    coalesce(current_setting('app.via_rpc', true) = 'true', false),
    nullif(current_setting('app.rpc_name', true), ''),
    p_op, p_table, p_pk, p_before, p_after
  );
END;
$$;

-- generic trigger function for protected tables (used in A4)
CREATE OR REPLACE FUNCTION public.trg_protected_write_audit() RETURNS trigger
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_pk text;
  v_pk_col text := COALESCE(TG_ARGV[0], 'id');
BEGIN
  IF TG_OP = 'DELETE' THEN
    EXECUTE format('SELECT ($1).%I::text', v_pk_col) INTO v_pk USING OLD;
  ELSE
    EXECUTE format('SELECT ($1).%I::text', v_pk_col) INTO v_pk USING NEW;
  END IF;

  PERFORM public.audit_log_write(
    TG_OP,
    TG_TABLE_NAME,
    v_pk,
    CASE WHEN TG_OP IN ('UPDATE', 'DELETE') THEN to_jsonb(OLD) END,
    CASE WHEN TG_OP IN ('UPDATE', 'INSERT') THEN to_jsonb(NEW) END
  );

  -- NOTE: caller_role is resolved via user_profiles join inside audit_log_write.
  -- service_role / cron contexts will record caller_role = NULL (no auth.uid()).

  RETURN COALESCE(NEW, OLD);
END;
$$;

REVOKE EXECUTE ON FUNCTION public.audit_log_write FROM PUBLIC;
REVOKE EXECUTE ON FUNCTION public.trg_protected_write_audit() FROM PUBLIC;

COMMIT;

-- A4: Install write-audit triggers on protected tables
-- Constitutional article enforced: 8 (every write recorded)

BEGIN;

DROP TRIGGER IF EXISTS machines_write_audit ON public.machines;
CREATE TRIGGER machines_write_audit
  AFTER INSERT OR UPDATE OR DELETE ON public.machines
  FOR EACH ROW EXECUTE FUNCTION public.trg_protected_write_audit('machine_id');

DROP TRIGGER IF EXISTS shelf_configurations_write_audit ON public.shelf_configurations;
CREATE TRIGGER shelf_configurations_write_audit
  AFTER INSERT OR UPDATE OR DELETE ON public.shelf_configurations
  FOR EACH ROW EXECUTE FUNCTION public.trg_protected_write_audit('shelf_id');

DROP TRIGGER IF EXISTS planogram_write_audit ON public.planogram;
CREATE TRIGGER planogram_write_audit
  AFTER INSERT OR UPDATE OR DELETE ON public.planogram
  FOR EACH ROW EXECUTE FUNCTION public.trg_protected_write_audit('planogram_id');

DROP TRIGGER IF EXISTS sim_cards_write_audit ON public.sim_cards;
CREATE TRIGGER sim_cards_write_audit
  AFTER INSERT OR UPDATE OR DELETE ON public.sim_cards
  FOR EACH ROW EXECUTE FUNCTION public.trg_protected_write_audit('sim_id');

-- A5: Tag canonical RPCs with app.via_rpc and app.rpc_name
-- Pattern (apply to each canonical RPC):

-- Pattern: prepend two lines to each canonical RPC body. NO behavior change.
-- Live signature verified: toggle_machine_refill(p_machine_name text, p_include boolean) returns jsonb

CREATE OR REPLACE FUNCTION public.toggle_machine_refill(
  p_machine_name text,
  p_include      boolean
) RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = public
AS $$
DECLARE
  v_result jsonb;
BEGIN
  -- ── Phase A tagging (the only Phase A change) ──
  PERFORM set_config('app.via_rpc', 'true', true);
  PERFORM set_config('app.rpc_name', 'toggle_machine_refill', true);

  -- ── existing body (unchanged) ──
  UPDATE public.machines
     SET include_in_refill = p_include
   WHERE official_name = p_machine_name
   RETURNING jsonb_build_object(
     'machine_id', machine_id,
     'machine_name', official_name,
     'include_in_refill', include_in_refill
   ) INTO v_result;

  RETURN v_result;
  -- (the trigger from A4 records the UPDATE in write_audit_log;
  --  via_rpc=true distinguishes this from a direct .update() in the FE)
END;
$$;

-- Phase A patches all canonical writers (verified inventory: 25 functions).
-- Role validation per Article 4 is Phase B work — current RPCs have no role guards;
-- the Next.js middleware enforces role at the route level today.
Verification
After deploying, run a UI action that calls the RPC, then query write_audit_log for that PK — the row should have via_rpc = true and rpc_name populated. A direct .update() from the FE would log via_rpc = false, which is exactly the signal Phase B needs.

# .github/workflows/governance.yml
name: Governance
on: pull_request

jobs:
  protected-write-lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Detect direct writes to protected tables
        run: bash scripts/governance/protected_write_lint.sh
        env:
          PROTECTED_TABLES: "machines shelf_configurations planogram sim_cards sales settlements pod_inventory_audit_log warehouse_inventory"
          FAIL_ON_VIOLATION: "false"  # warn mode in Phase A; flip to true per-entity in Phase B

  rls-coverage:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Apply migrations to ephemeral PG
        run: scripts/governance/apply_to_ephemeral_pg.sh
      - name: Check RLS on protected tables
        run: psql "$EPHEMERAL_DB" -f scripts/governance/rls_coverage.sql

  rpc-audit-coverage:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Apply migrations to ephemeral PG
        run: scripts/governance/apply_to_ephemeral_pg.sh
      - name: Check every DEFINER calls audit_log_write
        run: psql "$EPHEMERAL_DB" -f scripts/governance/rpc_audit_coverage.sql

#!/usr/bin/env bash
# scripts/governance/protected_write_lint.sh
set -e

VIOLATIONS=0
for table in $PROTECTED_TABLES; do
  matches=$(grep -rEn "supabase\.from\(\"$table\"\)\.(update|insert|delete)" src/ \
    --exclude-dir=node_modules --include='*.ts' --include='*.tsx' || true)
  if [ -n "$matches" ]; then
    echo "::warning::Direct write to protected table '$table' detected:"
    echo "$matches"
    VIOLATIONS=$((VIOLATIONS + 1))
  fi
done

if [ "$VIOLATIONS" -gt "0" ] && [ "$FAIL_ON_VIOLATION" = "true" ]; then
  echo "$VIOLATIONS protected-table violation(s) found. Use the canonical RPC."
  exit 1
fi

Cutover Plan

  1. Branch. Create a Supabase preview branch from main (Supabase MCP create_branch). All A1–A5 migrations apply there first.
  2. Apply A1 → A4 in order. Each is a separate migration file. Verify SELECT count(*) FROM write_audit_log climbs as soon as A4 lands.
  3. Smoke test. Trigger one machine update from /admin/machines; one repurpose from /field/config/machines; one stock refresh. Check write_audit_log on the branch — every row should appear, with via_rpc matching expectation.
  4. Apply A5 (RPC tagging). Re-run the same smoke tests; via_rpc = true should now appear for the canonical RPC paths and remain false for the direct-update paths. This is the signal that fuels Phase B planning.
  5. Merge branch to prod via Supabase MCP merge_branch. Migrations apply transactionally.
  6. Deploy A6 + A7 (CI + repo docs) as a single PR. Warn mode means it cannot break a merge.
  7. Observe for 48 hours. Daily check: SELECT table_name, count(*) FROM write_audit_log WHERE at > now() - interval '24h' AND via_rpc = false GROUP BY 1; — this number is the Phase B prioritization signal.

What Phase B looks like

Once Phase A is stable for 48 hours, Phase B picks one entity at a time and migrates the FE to the canonical RPC. Order: include_in_refill toggle (smallest, two call sites) → machine status field → shelf configurations → planogram editor → sim_cards. Each entity gets one PR. Each PR flips FAIL_ON_VIOLATION=true for its entity in the CI lint, sealing the perimeter behind it. By the end of Phase B, the FE has one path per entity, the audit log has full RPC tagging, and the FAIL_ON_VIOLATION flag is true for every protected entity.

Phase C is the FE consolidation — merging /app/admin/* and /field/config/* into shared modules now that the write contract is sealed.
